DROP TABLE IF EXISTS router_db;

CREATE TABLE router_db (
    [packet id]                       PRIMARY KEY,
    [src ip]             CHAR (7, 15),
    [dst ip]             CHAR (7, 15),
    [request type]       CHAR,
    [request parameters] CHAR,
    [packet headers]     CHAR,
    data                 CHAR,
    [src port]           DECIMAL      NOT NULL,
    [dst port]           DECIMAL      NOT NULL,
    [src mac]            CHAR,
    [dst mac]            CHAR
);


-- Index: sqlite_autoindex_router_db_1
DROP INDEX IF EXISTS sqlite_autoindex_router_db_1;

CREATE INDEX sqlite_autoindex_router_db_1 ON router_db (
    "packet id" COLLATE BINARY
);


COMMIT TRANSACTION;
PRAGMA foreign_keys = on;
